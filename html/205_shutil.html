shutils
The shutil module offers a number of high-level operations on files and collections of files.
In particular, functions are provided which support file copying and removal

    import shutil
    import os
    from os import path

    dir()[-3:]
    //    ['os', 'path', 'shutil']

    archive_name = path.join(path.realpath('.'),'myarchive')
    archive_name

    root_dir= path.join(path.realpath('.'), 'rooty')

    # from temp directory
    shutil.make_archive(archive_name, 'gztar', root_dir)
