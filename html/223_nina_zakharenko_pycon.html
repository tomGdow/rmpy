Making a class Iterable
          In order for a class to be iterable, a class needs to implement
          __iter__()

          __iter__() must return an iterator 

          In order to be an iterator, a class needs to implement
          __next__() which MUST raise StopIteration when there are
          no more items to return
Basic Example
 
    class TomIter:
        def __init__(self, start, stop):
            self.start = start
            self.stop = stop
            self.cursor = start

        def __iter__(self):
            return self

        def __next__(self):
            if self.cursor >= self.stop:
                raise StopIteration
            cursor_at = self.cursor
            self.cursor += 1
            return cursor_at

    for x in TomIter(1,10):
        print(f"{x} -> {x*x}") 

    //  1 -> 1
    //  2 -> 4
    //  3 -> 9
    //  4 -> 16
    //  5 -> 25
    //  6 -> 36
    //  7 -> 49
    //  8 -> 64
    //  9 -> 81
Use a Generator Instead

      def mygen(start,stop):
          cursor_at = start
          while cursor_at < stop:
           yield cursor_at
           cursor_at += 1

      class TomIter:
          def __init__(self, start, stop):
              self.start  = start
              self.stop = stop
             
          def mygen(self):
          
              cursor_at = self.start
              while cursor_at < self.stop:
                  yield cursor_at
                  cursor_at += 1

      t = TomIter(10,20)
      x = TomIter(20,30) 

      for (x,y)  in zip(t.mygen(), x.mygen()):
          print(f"{x} {y}")
      
      //  10 20
      //  11 21
      //  12 22
