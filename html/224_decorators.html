Decorator (Inheritance) 

    class Vehicle:
        def __init__(self, type):
            self.type = type

        def addTwo(self, a,b):
            return a + b

    class Car(Vehicle): 
        def __init__(self, v):
            super().__init__(v)

        def fix(func):    
            def inner(self,a,b):
                return func(self,int(a), int(b)) 
            return inner
        
        @fix 
        def addTwo(self,a,b):
            return super().addTwo(a,b)


    mycar = Car('toyota')
    print(mycar.type)                                       //  toyota
    print(mycar.addTwo(2,3))                                //  5
    print(Vehicle.addTwo('xyz', '2', '3'))                  // 23
    print(mycar.__class__.__mro__[1].addTwo("xyz","2","3")  // 23
    print(mycar.__class__.__mro__)

Decorator (Composition) 

    class Engine:
        def addTwo(self, a, b):
            return a + b


    class Lorry:
        def __init__(self, type):
            self.type = type
            self.fns=Engine()

        def fix(func):
            def inner(self, a, b):
                return func(self, int(a), int(b)) 
            return inner

        @fix
        def addTwo(self, a,b):
            return self.fns.addTwo(a,b)

    mylorry = Lorry('ford')
    print(mylorry.type)                             //  ford
    print(mylorry.fns.addTwo(2,3))                  //  5
    print(mylorry.fns.addTwo('2', '3'))             //  23
    print(mylorry.addTwo('2', '3'))                 //  5


    print(mylorry.__class__) 
                            //  <class '__main__.Lorry'>

    print(mylorry.__class__.__mro__)
                           //   (
                           //     <class '__main__.Lorry'>, 
                           //     <class 'object'>
                           //   )
