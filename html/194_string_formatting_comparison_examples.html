Write an email address

    fname="John"
    lname="Doe"
    middle_initial="o"
    email_host="gmail"
    domain_name="com"

    # f-string

    print(f"{fname.lower()}\
    {middle_initial.lower()}\
    {lname.lower()}\
    \N{COMMERCIAL AT}\
    {email_host.lower()}\
    .{domain_name.lower()}")

    //    johnodoe@gmail.com

    # .format() method

    str2="{}{}{}\N{COMMERCIAL AT}{}.{}"

    str2.format(fname.lower(),\
    middle_initial.lower(),\
    lname.lower(),\
    email_host.lower(),\
    domain_name.lower())

    //    'johnodoe@gmail.com'

    str3="{fname}{middle_initial}{lname}\N{COMMERCIAL AT}{email_host}.{domain_name}"

    str3.format(fname=fname.lower(),\ 
    middle_initial=middle_initial.lower(),\
    lname=lname.lower(),\ 
    email_host=email_host.lower(),\ 
    domain_name=domain_name.lower())

    //    'johnodoe@gmail.com'    // Note no inverted commas around fname etc

      # % method

     print("%s%s%s\N{COMMERCIAL AT}%s%s" % (
     fname.lower(), 
     middle_initial.lower(), 
     lname.lower(), 
     email_host.lower(), 
     domain_name.lower()))

     //    johnodoe@gmailcom

     print("%(fname)s\
            %(middle_initial)s\
            %(lname)s\
            \N{COMMERCIAL AT}\
            %(email_host)s\
            %(domain_name)s" % {'fname': fname.lower(), 
                                'middle_initial':middle_initial.lower(),
                                'lname':lname.lower(), 
                                'email_host':email_host.lower(), 
                                'domain_name':domain_name.lower()})

     //    johnodoe@gmailcom
     //     Note the inverted commas around 'fname' etc in the dictionary keys

     # Template method

     from string import Template
     email_template=Template("$fname$middle_initial$lname\N{COMMERCIAL AT}$email_host.$domain_name")

     email_template.substitute(
     fname=fname.lower(), 
     lname=lname.lower(), 
     middle_initial=middle_initial.lower(),
     email_host=email_host.lower(),
     domain_name=domain_name.lower())

     //    'johnodoe@gmail.com'
Format a million
    
    mymill=1000000

    print("{0:,}".format(mymill))
    //    1,000,000

    print(f"{mymill:,}")
    //    1,000,000
