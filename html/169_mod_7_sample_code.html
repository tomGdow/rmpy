    # Nested Data Structures
    # difference between append and extend
    # **warning** under most circumstances your data structures should only
    # contain a single data type

    a = ['a', 'b', 'c']
    print(a)

    # append will add a sequence as a *single element* of the list
    a.append(('d', 'e'))
    print(a)        //    ['a', 'b', 'c', ('d', 'e')]
    a.append(['f', 'g'])
    print(a)        //    ['a', 'b', 'c', ('d', 'e'), ['f', 'g']]
    print(len(a))   //    5

    # with extend, you can pass in a sequence and it
    # will add each element of the sequence as a seperate list element
    a.extend(('h', 'i'))
    print(a)        //    ['a', 'b', 'c', ('d', 'e'), ['f', 'g'], 'h', 'i']

    a.extend(['j', 'k'])
    print(a)        //    ['a', 'b', 'c', ('d', 'e'), ['f', 'g'], 'h', 'i', 'j', 'k']
    print(len(a))   //    9

    # accessing all cars in the multi-story garage
    for floor in lot_3d:    # cycle through each floor in the multi-story garage
        for row in floor:   # cycle through each row in the floor
            for car in row: # cycle through each car in the row
                print(car)
