     (5)  Integers 

            def incr(x):
               return x+1

            def ONE(f):
                def inner(x):
                    return f(x)
                return inner
    
            //  ONE(incr)(0)
            //  1

            def TWO(f):
                  def inner(x):
                    return f(f(x))

     (6)  SUCC  (Successor function)

          def SUCC(n):
              def inner(f):
                  def center(x):
                      return f(n(f)(x))
                  return center
              return inner
   
          //  SUCC(FOUR)(incr)(0)
          //  5

          def NOSUCC(n):
              def inner(f):
                  def center(x):
                      return n(f)(x)
                  return center
              return inner
   
          //  NOSUCC(FOUR)(incr)(0)
          //  4
          
          LAMBDA

          xSUCC   = lambda n: lambda f: lambda x: f(n(f)(x))
          xNOSUCC = lambda n: lambda f: lambda x:   n(f)(x)

     (8)  MULTIPLY

          def MUL(x):
             def inner(y):
                 def center(z):
                     return y(x(z))
                 return center
             return inner
